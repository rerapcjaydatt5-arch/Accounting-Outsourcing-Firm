<script>
    // Staff Cost comparison calculation logic
    const slider = document.getElementById('staff-slider');
    const staffValText = document.getElementById('staff-val');
    const savingsDisplay = document.getElementById('savings-display');

    if (slider && savingsDisplay && staffValText) {
        slider.addEventListener('input', function() {
            const val = parseInt(slider.value);
            // Staff label plural helper
            staffValText.innerText = val === 1 ? '1 Staff' : val + ' Staff';
            
            // Typical AUD wage gap per outsourced resource (~$70,000 savings annually)
            const calculatedSavings = val * 70000;
            savingsDisplay.innerText = '$' + calculatedSavings.toLocaleString();
        });
    }

    // Tab switcher script for financial service sectors
    function switchService(serviceId) {
        // Hide all blocks
        const blocks = document.querySelectorAll('.service-content-block');
        blocks.forEach(block => block.classList.add('hidden'));

        // Show targets block
        const activeBlock = document.getElementById('service-' + serviceId);
        if (activeBlock) {
            activeBlock.classList.remove('hidden');
        }

        // De-activate all navigation tab buttons
        const tabs = document.querySelectorAll('.service-tab-btn');
        tabs.forEach(tab => {
            tab.className = "service-tab-btn px-5 py-3 rounded-full text-xs font-bold border border-apple-border transition-all duration-300 hover:bg-apple-lightGray text-apple-darkGray";
        });

        // Activate matching selected tab
        const currentActiveBtn = document.getElementById('btn-' + serviceId);
        if (currentActiveBtn) {
            currentActiveBtn.className = "service-tab-btn active px-5 py-3 rounded-full text-xs font-bold border border-apple-border transition-all duration-300 bg-apple-black text-white";
        }
    }

    // Toggle Mobile Drawer menu state
    const mobileMenu = document.getElementById('mobile-menu');
    const menuToggle = document.getElementById('menu-toggle');
    const menuClose = document.getElementById('menu-close');

    if (menuToggle && mobileMenu) {
        menuToggle.addEventListener('click', () => {
            mobileMenu.classList.remove('translate-x-full');
        });
    }

    if (menuClose && mobileMenu) {
        menuClose.addEventListener('click', () => {
            mobileMenu.classList.add('translate-x-full');
        });
    }

    function toggleMobileMenu() {
        if (mobileMenu) {
            mobileMenu.classList.add('translate-x-full');
        }
    }

    // Contact registration form handling with custom visual confirmation
    const trialForm = document.getElementById('trial-form');
    const trialSuccessBlock = document.getElementById('trial-form-success');

    if (trialForm && trialSuccessBlock) {
        trialForm.addEventListener('submit', function(e) {
            e.preventDefault(); // Block standard form refresh actions
            
            if (trialForm.checkValidity()) {
                trialSuccessBlock.classList.remove('hidden');
            }
        });
    }

    function resetTrialForm() {
        if (trialForm && trialSuccessBlock) {
            trialForm.reset();
            trialSuccessBlock.classList.add('hidden');
        }
    }

    // Scroll header scroll class modifier
    window.addEventListener('scroll', function() {
        const header = document.querySelector('header');
        if (header) {
            if (window.scrollY > 40) {
                header.classList.add('shadow-sm');
                header.style.backgroundColor = 'rgba(255, 255, 255, 0.88)';
            } else {
                header.classList.remove('shadow-sm');
                header.style.backgroundColor = 'rgba(255, 255, 255, 0.72)';
            }
        }
    });

    /* Live Publishing Helper Functions */
    function togglePublishSuite() {
        const suite = document.getElementById('publish-suite');
        if (suite) {
            suite.classList.add('hidden');
        }
    }

    function toggleDeployModal() {
        const modal = document.getElementById('deploy-modal');
        if (modal) {
            modal.classList.toggle('hidden');
        }
    }

    function downloadProductionFile() {
        // Fetch full HTML document code
        const fullContent = "<!DOCTYPE html>\n" + document.documentElement.outerHTML;
        
        // Exclude the floating Publisher Assistant panel from exported production code so it remains pristine for your clients
        const cleanedContent = fullContent.replace(/[\s\S]*/, "");

        const blob = new Blob([cleanedContent], { type: "text/html;charset=utf-8" });
        const url = URL.createObjectURL(blob);
        const a = document.createElement("a");
        a.href = url;
        a.download = "index.html";
        document.body.appendChild(a);
        a.click();
        
        // Clean temporary pointers
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
    }
</script>
